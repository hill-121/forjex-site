class AgentSystem:
    # 多Agent系统核心代码
    def __init__(self):
        self.agents = [
            "采集菌",  # 周一：信息收集
            "文案猫",  # 周二：内容创作
            "画师鹰",  # 周三：视觉设计
            "动画鱼",  # 周四：视频制作
            "发布猴"   # 周五：内容发布
        ]
    
    def run_workflow(self, topic):
        for agent in self.agents:
            result = agent.execute(topic)
            print(f"{agent.name}: {result}")
