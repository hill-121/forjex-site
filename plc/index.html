tia_portal_ladder.py
from snap7 import client

# TIA Portal 梯形图编程示例
plc = client.Client()
plc.connect('192.168.0.1', 0, 1)

def read_input(address):
    return plc.read_area('I', address, 1)

def write_output(address, value):
    plc.write_area('Q', address, value)

# 梯形图逻辑: I0.0 --[ ]-- Q0.0
while True:
    if read_input(0) == 1:  # I0.0 接通
        write_output(0, 1)       # Q0.0 输出
